914	    func resetToFirstRun() {
915	        // Suppress all persistence for the rest of this process's life so the
916	        // wipe below isn't undone by a save (notably prepareForTermination on
917	        // the NSApp.terminate path).
918	        isResettingToFirstRun = true
919	        // 1) Stop live agent terminals + their persistent screen sessions so a
920	        //    fresh launch starts clean and nothing is left running unattended.
921	        //    Non-destructive to the agents' histories (those live with their
922	        //    harness); the process can be relaunched and resumed afterward.
923	        for entry in state.processEntries where activeSessions[entry.id] != nil {
924	            terminate(entry)
925	        }
926	        Self.killAllPersistentScreens()
927	
928	        // 2) Back up + remove the workspace state (so the next launch bootstraps
929	        //    fresh — the bootstrapper treats a missing file as first run) and
930	        //    clear *all* Workbench preferences for a true factory state, not a
931	        //    half-reset. This data wipe is unit-tested via `WorkbenchFactoryReset`.
932	        WorkbenchFactoryReset.resetToFactoryDefaults(
933	            stateURL: paths.stateURL,
934	            defaults: .standard,
935	            defaultsDomain: WorkbenchRelease.bundleIdentifier,
936	            timestamp: Date()
937	        )
938	        UserDefaults.standard.synchronize()
939	
940	        // 3) Relaunch a fresh instance once this one exits, then quit.
941	        Self.relaunchAfterExit()

10560	            let workItem = DispatchWorkItem { [weak self, weak terminal] in

10563	                      terminal.superview === self else {
10564	                    return

10567	                // is in the alternate-screen buffer — i.e. a full-screen TUI
10568	                // (Claude Code, Codex, vim) where Ctrl-L means "repaint" and
10569	                // is harmless. In the normal buffer (a plain shell sitting at
